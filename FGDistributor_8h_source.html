122  class PropValPair {
123  public:
124  PropValPair(std::string prop, std::string val, FGPropertyManager* propMan) {
125  // Process property to be set
126  PropMan = propMan;
127  sign = 1;
128  FGPropertyNode *node = propMan->GetNode(prop, false);
129  if (node) PropNode = node;
130  else PropNode = 0;
131  PropName = prop;
132 
133  // Process set value
134  LateBoundValue = false;
135  Val = 0;
136  ValString = val;
137  if (is_number(ValString)) {
138  Val = new FGRealValue(atof(ValString.c_str()));
139  } else {
140  // "value" must be a property if execution passes to here.
141  if (ValString[0] == '-') {
142  sign = -1;
143  ValString.erase(0,1);
144  }
145  node = propMan->GetNode(ValString, false); // Reuse the node variable
146  if (node) Val = new FGPropertyValue(node);
147  else {
148  Val = new FGPropertyValue(ValString, propMan);
149  LateBoundValue = true;
150  }
151  }
152  }
153 
154  ~PropValPair() {
155  delete PropNode;
156  delete PropMan;
157  }
158 
159  void SetPropToValue() {
160  if (PropNode == 0) {
161  if (PropMan->HasNode(PropName)) {
162  PropNode = PropMan->GetNode(PropName);
163  } else {
164  throw(PropName+" in distributor component is not known");
165  }
166  }
167  PropNode->setDoubleValue(Val->GetValue()*sign);
168  }
169 
170  std::string GetPropName() {return PropName;}
171  std::string GetValString() {return ValString;}
172  FGPropertyNode* GetPropNode() {return PropNode;}
173  bool GetLateBoundValue() {return LateBoundValue;}
174  private:
175  std::string PropName;
176  FGPropertyNode* PropNode;
177  FGPropertyManager* PropMan;
178  FGParameter* Val;
179  std::string ValString;
180  bool LateBoundValue;
181  int sign;
