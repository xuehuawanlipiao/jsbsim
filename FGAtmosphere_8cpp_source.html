57 // Atmosphere constants in British units converted from the SI values specified in the
58 // ISA document - https://ntrs.nasa.gov/archive/nasa/casi.ntrs.nasa.gov/19770009539.pdf

60 const double KtoDegR = 1.8; // Kelvin to degree Rankine
61 const double FGAtmosphere::Rstar = 8.31432 * (FGJSBBase::kgtoslug / (KtoDegR * FGJSBBase::fttom * FGJSBBase::fttom)); // ft*lbf/R/mol
62 const double FGAtmosphere::Mair = 28.9645 * FGJSBBase::kgtoslug / 1000.0; // slug/mol
63 const double FGAtmosphere::g0 = 9.80665 / FGJSBBase::fttom; // ft/s^2
64 double FGAtmosphere::Reng = Rstar / Mair;
65 
66 const double FGAtmosphere::SHRatio = 1.40;

68 const double FGAtmosphere::StdDaySLtemperature = 518.67;
69 const double FGAtmosphere::StdDaySLpressure = 2116.228;
70 const double FGAtmosphere::StdDaySLsoundspeed = sqrt(SHRatio*Reng*StdDaySLtemperature);

93 bool FGAtmosphere::InitModel(void)
94 {
95  if (!FGModel::InitModel()) return false;
96 
97  Calculate(0.0);
98  SLtemperature = Temperature = StdDaySLtemperature;
99  SLpressure = Pressure = StdDaySLpressure;
100  SLdensity = Density = Pressure/(Reng*Temperature);
101  SLsoundspeed = Soundspeed = StdDaySLsoundspeed;
102 
103  return true;
104 }

121 void FGAtmosphere::Calculate(double altitude)
122 {
123  FGPropertyNode* node = PropertyManager->GetNode();
124  if (!PropertyManager->HasNode("atmosphere/override/temperature"))
125  Temperature = GetTemperature(altitude);
126  else
127  Temperature = node->GetDouble("atmosphere/override/temperature");
128 
129  if (!PropertyManager->HasNode("atmosphere/override/pressure"))
130  Pressure = GetPressure(altitude);
131  else
132  Pressure = node->GetDouble("atmosphere/override/pressure");
133 
134  if (!PropertyManager->HasNode("atmosphere/override/density"))
135  Density = GetDensity(altitude);
136  else
137  Density = node->GetDouble("atmosphere/override/density");
138 
139  Soundspeed = sqrt(SHRatio*Reng*Temperature);
140  PressureAltitude = CalculatePressureAltitude(Pressure, altitude);
141  DensityAltitude = CalculateDensityAltitude(Density, altitude);
142 
143  Viscosity = Beta * pow(Temperature, 1.5) / (SutherlandConstant + Temperature);
144  KinematicViscosity = Viscosity / Density;
145 }

183 double FGAtmosphere::ConvertToRankine(double t, eTemperature unit) const
184 {
185  double targetTemp=0; // in degrees Rankine
186 
187  switch(unit) {
188  case eFahrenheit:
189  targetTemp = t + 459.67;
190  break;
191  case eCelsius:
192  targetTemp = (t + 273.15) * 1.8;
193  break;
194  case eRankine:
195  targetTemp = t;
196  break;
197  case eKelvin:
198  targetTemp = t*1.8;
199  break;
200  default:
201  break;
202  }
203 
204  return targetTemp;
205 }

235 double FGAtmosphere::ConvertToPSF(double p, ePressure unit) const
236 {
237  double targetPressure=0; // Pressure in PSF
238 
239  switch(unit) {
240  case ePSF:
241  targetPressure = p;
242  break;
243  case eMillibars:
244  targetPressure = p*2.08854342;
245  break;
246  case ePascals:
247  targetPressure = p*0.0208854342;
248  break;
249  case eInchesHg:
250  targetPressure = p*70.7180803;
251  break;
252  default:
253  throw("Undefined pressure unit given");
254  }
255 
256  return targetPressure;
257 }

303 // The bitmasked value choices are as follows:
304 // unset: In this case (the default) JSBSim would only print
305 // out the normally expected messages, essentially echoing
306 // the config files as they are read. If the environment
307 // variable is not set, debug_lvl is set to 1 internally
308 // 0: This requests JSBSim not to output any messages
309 // whatsoever.
310 // 1: This value explicity requests the normal JSBSim
311 // startup messages
312 // 2: This value asks for a message to be printed out when
313 // a class is instantiated
314 // 4: When this value is set, a message is displayed when a
315 // FGModel object executes its Run() method
316 // 8: When this value is set, various runtime state variables
317 // are printed out periodically
318 // 16: When set various parameters are sanity checked and
319 // a message is printed out when they go out of bounds
320 
321 void FGAtmosphere::Debug(int from)
322 {
323  if (debug_lvl <= 0) return;
324 
325  if (debug_lvl & 1) { // Standard console startup message output
326  if (from == 0) { // Constructor
327  }
328  }
329  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
330  if (from == 0) std::cout << "Instantiated: FGAtmosphere" << std::endl;
331  if (from == 1) std::cout << "Destroyed: FGAtmosphere" << std::endl;
332  }
333  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
334  }
335  if (debug_lvl & 8 ) { // Runtime state variables
336  }
337  if (debug_lvl & 16) { // Sanity checking
338  }
339  if (debug_lvl & 128) { //
340  }
341  if (debug_lvl & 64) {
342  if (from == 0) { // Constructor
343  }
344  }
345 }
