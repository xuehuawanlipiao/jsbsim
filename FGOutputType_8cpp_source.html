103 bool FGOutputType::Load(Element* element)
104 {
105  if (element->FindElementValue("simulation") == string("ON"))
106  SubSystems += ssSimulation;
107  if (element->FindElementValue("aerosurfaces") == string("ON"))
108  SubSystems += ssAerosurfaces;
109  if (element->FindElementValue("rates") == string("ON"))
110  SubSystems += ssRates;
111  if (element->FindElementValue("velocities") == string("ON"))
112  SubSystems += ssVelocities;
113  if (element->FindElementValue("forces") == string("ON"))
114  SubSystems += ssForces;
115  if (element->FindElementValue("moments") == string("ON"))
116  SubSystems += ssMoments;
117  if (element->FindElementValue("atmosphere") == string("ON"))
118  SubSystems += ssAtmosphere;
119  if (element->FindElementValue("massprops") == string("ON"))
120  SubSystems += ssMassProps;
121  if (element->FindElementValue("position") == string("ON"))
122  SubSystems += ssPropagate;
123  if (element->FindElementValue("coefficients") == string("ON") || element->FindElementValue("aerodynamics") == string("ON"))
124  SubSystems += ssAeroFunctions;
125  if (element->FindElementValue("ground_reactions") == string("ON"))
126  SubSystems += ssGroundReactions;
127  if (element->FindElementValue("fcs") == string("ON"))
128  SubSystems += ssFCS;
129  if (element->FindElementValue("propulsion") == string("ON"))
130  SubSystems += ssPropulsion;
131 
132  Element *property_element = element->FindElement("property");
133 
134  while (property_element) {
135  string property_str = property_element->GetDataLine();
136  FGPropertyNode* node = PropertyManager->GetNode(property_str);
137  if (!node) {
138  cerr << property_element->ReadFrom()
139  << fgred << highint << endl << " No property by the name "
140  << property_str << " has been defined. This property will " << endl
141  << " not be logged. You should check your configuration file."
142  << reset << endl;
143  } else {
144  if (property_element->HasAttribute("apply")) {
145  string function_str = property_element->GetAttributeValue("apply");
146  FGOutput* Output = FDMExec->GetOutput();
147  FGTemplateFunc* f = Output->GetTemplateFunc(function_str);
148  if (f)
149  OutputParameters.push_back(new FGFunctionValue(node, f));
150  else {
151  cerr << property_element->ReadFrom()
152  << fgred << highint << " No function by the name "
153  << function_str << " has been defined. This property will "
154  << "not be logged. You should check your configuration file."
155  << reset << endl;
156  }
157  }
158  else
159  OutputParameters.push_back(new FGPropertyValue(node));
160 
161  if (property_element->HasAttribute("caption"))
162  OutputCaptions.push_back(property_element->GetAttributeValue("caption"));
163  else
164  OutputCaptions.push_back("");
165  }
166  property_element = element->FindNextElement("property");
167  }
168 
169  double outRate = 1.0;
170  if (element->HasAttribute("rate"))
171  outRate = element->GetAttributeValueAsNumber("rate");
172 
173  SetRateHz(outRate);
174 
175  return true;
176 }

206 void FGOutputType::SetRateHz(double rtHz)
207 {
208  rtHz = rtHz>1000?1000:(rtHz<0?0:rtHz);
209  if (rtHz > 0) {
210  SetRate(0.5 + 1.0/(FDMExec->GetDeltaT()*rtHz));
211  Enable();
212  } else {
213  SetRate(1);
214  Disable();
215  }
216 }

236 // The bitmasked value choices are as follows:
237 // unset: In this case (the default) JSBSim would only print
238 // out the normally expected messages, essentially echoing
239 // the config files as they are read. If the environment
240 // variable is not set, debug_lvl is set to 1 internally
241 // 0: This requests JSBSim not to output any messages
242 // whatsoever.
243 // 1: This value explicity requests the normal JSBSim
244 // startup messages
245 // 2: This value asks for a message to be printed out when
246 // a class is instantiated
247 // 4: When this value is set, a message is displayed when a
248 // FGModel object executes its Run() method
249 // 8: When this value is set, various runtime state variables
250 // are printed out periodically
251 // 16: When set various parameters are sanity checked and
252 // a message is printed out when they go out of bounds
253 
254 void FGOutputType::Debug(int from)
255 {
256  if (debug_lvl <= 0) return;
257 
258  if (debug_lvl & 1) { // Standard console startup message output
259  if (from == 0) { // Constructor
260 
261  }
262  if (from == 2) {
263  if (SubSystems & ssSimulation) cout << " Simulation parameters logged" << endl;
264  if (SubSystems & ssAerosurfaces) cout << " Aerosurface parameters logged" << endl;
265  if (SubSystems & ssRates) cout << " Rate parameters logged" << endl;
266  if (SubSystems & ssVelocities) cout << " Velocity parameters logged" << endl;
267  if (SubSystems & ssForces) cout << " Force parameters logged" << endl;
268  if (SubSystems & ssMoments) cout << " Moments parameters logged" << endl;
269  if (SubSystems & ssAtmosphere) cout << " Atmosphere parameters logged" << endl;
270  if (SubSystems & ssMassProps) cout << " Mass parameters logged" << endl;
271  if (SubSystems & ssAeroFunctions) cout << " Coefficient parameters logged" << endl;
272  if (SubSystems & ssPropagate) cout << " Propagate parameters logged" << endl;
273  if (SubSystems & ssGroundReactions) cout << " Ground parameters logged" << endl;
274  if (SubSystems & ssFCS) cout << " FCS parameters logged" << endl;
275  if (SubSystems & ssPropulsion) cout << " Propulsion parameters logged" << endl;
276  if (!OutputParameters.empty()) cout << " Properties logged:" << endl;
277  for (unsigned int i=0;i<OutputParameters.size();i++) {
278  cout << " - " << OutputParameters[i]->GetName() << endl;
279  }
280  }
281  }
282  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
283  if (from == 0) cout << "Instantiated: FGOutputType" << endl;
284  if (from == 1) cout << "Destroyed: FGOutputType" << endl;
285  }
286  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
287  }
288  if (debug_lvl & 8 ) { // Runtime state variables
289  }
290  if (debug_lvl & 16) { // Sanity checking
291  }
292  if (debug_lvl & 64) {
293  if (from == 0) { // Constructor
294  }
295  }
296 }
