77 FGSwitch::FGSwitch(FGFCS* fcs, Element* element) : FGFCSComponent(fcs, element)
78 {
79  string value;
80  struct test *current_test;
81  Element *test_element;
82 
83  FGFCSComponent::bind(); // Bind() this component here in case it is used
84  // in its own definition for a sample-and-hold
85 
86  int num = element->GetNumElements("test");
87  if (element->FindElement("default")) num++;
88  tests.reserve(num);
89 
90  test_element = element->FindElement("default");
91  if (test_element) {
92  current_test = new struct test;
93  current_test->condition = 0;
94  value = test_element->GetAttributeValue("value");
95  current_test->setTestValue(value, Name, PropertyManager);
96  current_test->Default = true;
97  if (delay > 0 && is_number(value)) { // If there is a delay, initialize the
98  for (unsigned int i=0; i<delay-1; i++) { // delay buffer to the default value
99  output_array[i] = atof(value.c_str()); // for the switch if that value is a number.
100  }
101  }
102  tests.push_back(current_test);
103  }
104 
105  test_element = element->FindElement("test");
106  while (test_element) {
107  current_test = new struct test;
108  current_test->condition = new FGCondition(test_element, PropertyManager);
109  value = test_element->GetAttributeValue("value");
110  current_test->setTestValue(value, Name, PropertyManager);
111  tests.push_back(current_test);
112  test_element = element->FindNextElement("test");
113  }
114 
115  Debug(0);
116 }

133 bool FGSwitch::Run(void )
134 {
135  bool pass = false;
136  double default_output=0.0;
137 
138  // To detect errors early, make sure all conditions and values can be
139  // evaluated in the first time step.
140  if (!initialized) {
141  initialized = true;
142  VerifyProperties();
143  }
144 
145  for (unsigned int i=0; i<tests.size(); i++) {
146  if (tests[i]->Default) {
147  default_output = tests[i]->GetValue();
148  } else {
149  pass = tests[i]->condition->Evaluate();
150  }
151 
152  if (pass) {
153  Output = tests[i]->GetValue();
154  break;
155  }
156  }
157 
158  if (!pass) Output = default_output;
159 
160  if (delay != 0) Delay();
161  Clip();
162  if (IsOutput) SetOutput();
163 
164  return true;
165 }

179 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
180 // The bitmasked value choices are as follows:
181 // unset: In this case (the default) JSBSim would only print
182 // out the normally expected messages, essentially echoing
183 // the config files as they are read. If the environment
184 // variable is not set, debug_lvl is set to 1 internally
185 // 0: This requests JSBSim not to output any messages
186 // whatsoever.
187 // 1: This value explicity requests the normal JSBSim
188 // startup messages
189 // 2: This value asks for a message to be printed out when
190 // a class is instantiated
191 // 4: When this value is set, a message is displayed when a
192 // FGModel object executes its Run() method
193 // 8: When this value is set, various runtime state variables
194 // are printed out periodically
195 // 16: When set various parameters are sanity checked and
196 // a message is printed out when they go out of bounds

198 void FGSwitch::Debug(int from)
199 {
200  string comp, scratch;
201  string indent = " ";
202  //bool first = false;
203 
204  if (debug_lvl <= 0) return;
205 
206  if (debug_lvl & 1) { // Standard console startup message output
207  if (from == 0) { // Constructor
208  for (unsigned int i=0; i<tests.size(); i++) {
209  if (tests[i]->Default) {
210  if (tests[i]->OutputProp == 0) {
211  cout << " Switch default value is: " << tests[i]->OutputVal;
212  } else {
213  cout << " Switch default value is: " << tests[i]->OutputProp->GetName();
214  }
215  } else {
216  if (tests[i]->OutputProp == 0) {
217  cout << " Switch takes test " << i << " value (" << tests[i]->OutputVal << ")" << endl;
218  } else {
219  cout << " Switch takes test " << i << " value (" << tests[i]->OutputProp->GetName() << ")" << endl;
220  }
221  tests[i]->condition->PrintCondition(" ");
222  }
223  cout << endl;
224  }
225  if (IsOutput) {
226  for (unsigned int i=0; i<OutputNodes.size(); i++)
227  cout << " OUTPUT: " << OutputNodes[i]->getName() << endl;
228  }
229  }
230  }
231  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
232  if (from == 0) cout << "Instantiated: FGSwitch" << endl;
233  if (from == 1) cout << "Destroyed: FGSwitch" << endl;
234  }
235  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
236  }
237  if (debug_lvl & 8 ) { // Runtime state variables
238  }
239  if (debug_lvl & 16) { // Sanity checking
240  }
241  if (debug_lvl & 64) {
242  if (from == 0) { // Constructor
243  }
244  }
245 }
