133 bool FGAuxiliary::Run(bool Holding)
134 {
135  if (FGModel::Run(Holding)) return true; // return true if error returned from base class
136  if (Holding) return false;
137 
138  // Rotation
139 
140  vEulerRates(eTht) = in.vPQR(eQ)*in.CosPhi - in.vPQR(eR)*in.SinPhi;
141  if (in.CosTht != 0.0) {
142  vEulerRates(ePsi) = (in.vPQR(eQ)*in.SinPhi + in.vPQR(eR)*in.CosPhi)/in.CosTht;
143  vEulerRates(ePhi) = in.vPQR(eP) + vEulerRates(ePsi)*in.SinTht;
144  }
145 
146  // Combine the wind speed with aircraft speed to obtain wind relative speed
147  vAeroPQR = in.vPQR - in.TurbPQR;
148  vAeroUVW = in.vUVW - in.Tl2b * in.TotalWindNED;
149 
150  alpha = beta = adot = bdot = 0;
151  double AeroU2 = vAeroUVW(eU)*vAeroUVW(eU);
152  double AeroV2 = vAeroUVW(eV)*vAeroUVW(eV);
153  double AeroW2 = vAeroUVW(eW)*vAeroUVW(eW);
154  double mUW = AeroU2 + AeroW2;
155 
156  double Vt2 = mUW + AeroV2;
157  Vt = sqrt(Vt2);
158 
159  if ( Vt > 0.001 ) {
160  if (vAeroUVW(eW) != 0.0)
161  alpha = AeroU2 > 0.0 ? atan2(vAeroUVW(eW), vAeroUVW(eU)) : 0.0;
162  if (vAeroUVW(eV) != 0.0)
163  beta = mUW > 0.0 ? atan2(vAeroUVW(eV), sqrt(mUW)) : 0.0;
164 
165  //double signU=1;
166  //if (vAeroUVW(eU) < 0.0) signU=-1;
167 
168  if ( mUW >= 0.001 ) {
169  double Vtdot = (vAeroUVW(eU)*in.vUVWdot(eU) + vAeroUVW(eV)*in.vUVWdot(eV) + vAeroUVW(eW)*in.vUVWdot(eW))/Vt;
170  adot = (vAeroUVW(eU)*in.vUVWdot(eW) - vAeroUVW(eW)*in.vUVWdot(eU))/mUW;
171  // bdot = (signU*mUW*in.vUVWdot(eV)
172  // - vAeroUVW(eV)*(vAeroUVW(eU)*in.vUVWdot(eU) + vAeroUVW(eW)*in.vUVWdot(eW)))/(Vt2*sqrt(mUW));
173  bdot = (in.vUVWdot(eV)*Vt - vAeroUVW(eV)*Vtdot)/(Vt*sqrt(mUW));
174  }
175  }
176 
177  UpdateWindMatrices();
178 
179  Re = Vt * in.Wingchord / in.KinematicViscosity;
180 
181  double densityD2 = 0.5*in.Density;
182 
183  qbar = densityD2 * Vt2;
184  qbarUW = densityD2 * (mUW);
185  qbarUV = densityD2 * (AeroU2 + AeroV2);
186  Mach = Vt / in.SoundSpeed;
187  MachU = vMachUVW(eU) = vAeroUVW(eU) / in.SoundSpeed;
188  vMachUVW(eV) = vAeroUVW(eV) / in.SoundSpeed;
189  vMachUVW(eW) = vAeroUVW(eW) / in.SoundSpeed;
190 
191  // Position
192 
193  Vground = sqrt( in.vVel(eNorth)*in.vVel(eNorth) + in.vVel(eEast)*in.vVel(eEast) );
194 
195  psigt = atan2(in.vVel(eEast), in.vVel(eNorth));
196  if (psigt < 0.0) psigt += 2*M_PI;
197  gamma = atan2(-in.vVel(eDown), Vground);
198 
199  tat = in.Temperature*(1 + 0.2*Mach*Mach); // Total Temperature, isentropic flow
200  tatc = RankineToCelsius(tat);
201 
202  pt = PitotTotalPressure(Mach, in.Pressure);
203 
204  if (abs(Mach) > 0.0) {
205  vcas = VcalibratedFromMach(Mach, in.Pressure);
206  veas = sqrt(2 * qbar / in.DensitySL);
207  }
208  else
209  vcas = veas = 0.0;
210 
211  vPilotAccel.InitMatrix();
212  vNcg = in.vBodyAccel/in.SLGravity;
213  // Nz is Acceleration in "g's", along normal axis (-Z body axis)
214  Nz = -vNcg(eZ);
215  Ny = vNcg(eY);
216  vPilotAccel = in.vBodyAccel + in.vPQRidot * in.ToEyePt;
217  vPilotAccel += in.vPQRi * (in.vPQRi * in.ToEyePt);
218 
219  vNwcg = mTb2w * vNcg;
220  vNwcg(eZ) = 1.0 - vNwcg(eZ);
221 
222  vPilotAccelN = vPilotAccel / in.SLGravity;
223 
224  // VRP computation
225  vLocationVRP = in.vLocation.LocalToLocation( in.Tb2l * in.VRPBody );
226 
227  // Recompute some derived values now that we know the dependent parameters values ...
228  hoverbcg = in.DistanceAGL / in.Wingspan;
229 
230  FGColumnVector3 vMac = in.Tb2l * in.RPBody;
231  hoverbmac = (in.DistanceAGL + vMac(3)) / in.Wingspan;
232 
233  return false;
234 }

236 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
237 //
238 // From Stevens and Lewis, "Aircraft Control and Simulation", 3rd Ed., the
239 // transformation from body to wind axes is defined (where "a" is alpha and "B"
240 // is beta):
241 //
242 // cos(a)*cos(B) sin(B) sin(a)*cos(B)
243 // -cos(a)*sin(B) cos(B) -sin(a)*sin(B)
244 // -sin(a) 0 cos(a)
245 //
246 // The transform from wind to body axes is then,
247 //
248 // cos(a)*cos(B) -cos(a)*sin(B) -sin(a)
249 // sin(B) cos(B) 0
250 // sin(a)*cos(B) -sin(a)*sin(B) cos(a)
251 
252 void FGAuxiliary::UpdateWindMatrices(void)
253 {
254  double ca, cb, sa, sb;
255 
256  ca = cos(alpha);
257  sa = sin(alpha);
258  cb = cos(beta);
259  sb = sin(beta);
260 
261  mTw2b(1,1) = ca*cb;
262  mTw2b(1,2) = -ca*sb;
263  mTw2b(1,3) = -sa;
264  mTw2b(2,1) = sb;
265  mTw2b(2,2) = cb;
266  mTw2b(2,3) = 0.0;
267  mTw2b(3,1) = sa*cb;
268  mTw2b(3,2) = -sa*sb;
269  mTw2b(3,3) = ca;
270 
271  mTb2w = mTw2b.Transposed();
272 }

389 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
390 // The bitmasked value choices are as follows:
391 // unset: In this case (the default) JSBSim would only print
392 // out the normally expected messages, essentially echoing
393 // the config files as they are read. If the environment
394 // variable is not set, debug_lvl is set to 1 internally
395 // 0: This requests JSBSim not to output any messages
396 // whatsoever.
397 // 1: This value explicity requests the normal JSBSim
398 // startup messages
399 // 2: This value asks for a message to be printed out when
400 // a class is instantiated
401 // 4: When this value is set, a message is displayed when a
402 // FGModel object executes its Run() method
403 // 8: When this value is set, various runtime state variables
404 // are printed out periodically
405 // 16: When set various parameters are sanity checked and
406 // a message is printed out when they go out of bounds
407 
408 void FGAuxiliary::Debug(int from)
409 {
410  if (debug_lvl <= 0) return;
411 
412  if (debug_lvl & 1) { // Standard console startup message output
413  if (from == 0) { // Constructor
414 
415  }
416  }
417  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
418  if (from == 0) cout << "Instantiated: FGAuxiliary" << endl;
419  if (from == 1) cout << "Destroyed: FGAuxiliary" << endl;
420  }
421  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
422  }
423  if (debug_lvl & 8 ) { // Runtime state variables
424  }
425  if (debug_lvl & 16) { // Sanity checking
426  if (Mach > 100 || Mach < 0.00)
427  cout << "FGPropagate::Mach is out of bounds: " << Mach << endl;
428  if (qbar > 1e6 || qbar < 0.00)
429  cout << "FGPropagate::qbar is out of bounds: " << qbar << endl;
430  }
431  if (debug_lvl & 64) {
432  if (from == 0) { // Constructor
433  }
434  }
435 }
