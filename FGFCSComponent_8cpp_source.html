263 void FGFCSComponent::Clip(void)
264 {
265  if (clip) {
266  if (ClipMinPropertyNode != 0) clipmin = clipMinSign*ClipMinPropertyNode->getDoubleValue();
267  if (ClipMaxPropertyNode != 0) clipmax = clipMaxSign*ClipMaxPropertyNode->getDoubleValue();
268  if (Output > clipmax) Output = clipmax;
269  else if (Output < clipmin) Output = clipmin;
270  }
271 }

275 // The old way of naming FCS components allowed upper or lower case, spaces,
276 // etc. but then the names were modified to fit into a property name
277 // hierarchy. This was confusing (it wasn't done intentionally - it was a
278 // carryover from the early design). We now support the direct naming of

282 void FGFCSComponent::bind(void)
283 {
284  string tmp;
285  if (Name.find("/") == string::npos) {
286  tmp = "fcs/" + PropertyManager->mkPropertyName(Name, true);
287  } else {
288  tmp = Name;
289  }
290  PropertyManager->Tie( tmp, this, &FGFCSComponent::GetOutput);
291 }

293 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
294 // The bitmasked value choices are as follows:
295 // unset: In this case (the default) JSBSim would only print
296 // out the normally expected messages, essentially echoing
297 // the config files as they are read. If the environment
298 // variable is not set, debug_lvl is set to 1 internally
299 // 0: This requests JSBSim not to output any messages
300 // whatsoever.
301 // 1: This value explicity requests the normal JSBSim
302 // startup messages
303 // 2: This value asks for a message to be printed out when
304 // a class is instantiated
305 // 4: When this value is set, a message is displayed when a
306 // FGModel object executes its Run() method
307 // 8: When this value is set, various runtime state variables
308 // are printed out periodically
309 // 16: When set various parameters are sanity checked and
310 // a message is printed out when they go out of bounds
311 
312 void FGFCSComponent::Debug(int from)
313 {
314  if (debug_lvl <= 0) return;
315 
316  if (debug_lvl & 1) { // Standard console startup message output
317  if (from == 0) {
318  cout << endl << " Loading Component \"" << Name
319  << "\" of type: " << Type << endl;
320 
321  if (clip) {
322  string propsign;
323 
324  if (ClipMinPropertyNode != 0L) {
325  if (clipMinSign < 0.0) propsign="-";
326  cout << " Minimum limit: " << propsign << ClipMinPropertyNode->GetName() << endl;
327  } else {
328  cout << " Minimum limit: " << clipmin << endl;
329  }
330 
331  propsign="";
332 
333  if (ClipMaxPropertyNode != 0L) {
334  if (clipMaxSign < 0.0) propsign="-";
335  cout << " Maximum limit: " << propsign << ClipMaxPropertyNode->GetName() << endl;
336  } else {
337  cout << " Maximum limit: " << clipmax << endl;
338  }
339  }
340  if (delay > 0) cout <<" Frame delay: " << delay
341  << " frames (" << delay*dt << " sec)" << endl;
342  }
343  }
344  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
345  if (from == 0) cout << "Instantiated: FGFCSComponent" << endl;
346  if (from == 1) cout << "Destroyed: FGFCSComponent" << endl;
347  }
348  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
349  }
350  if (debug_lvl & 8 ) { // Runtime state variables
351  }
352  if (debug_lvl & 16) { // Sanity checking
353  }
354  if (debug_lvl & 64) {
355  if (from == 0) { // Constructor
356  }
357  }
358 }
