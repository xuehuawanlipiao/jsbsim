74 // (stolen from FGFS native_fdm.cxx)
75 // The function htond is defined this way due to the way some
76 // processors and OSes treat floating point values. Some will raise
77 // an exception whenever a "bad" floating point value is loaded into a
78 // floating point register. Solaris is notorious for this, but then
79 // so is LynxOS on the PowerPC. By translating the data in place,
80 // there is no need to load a FP register with the "corruped" floating
81 // point value. By doing the BIG_ENDIAN test, I can optimize the
82 // routine for big-endian processors so it can be as efficient as
83 // possible
84 static void htond (double &x)
85 {
86  if ( isLittleEndian ) {
87  int *Double_Overlay;
88  int Holding_Buffer;
89 
90  Double_Overlay = (int *) &x;
91  Holding_Buffer = Double_Overlay [0];
92 
93  Double_Overlay [0] = htonl (Double_Overlay [1]);
94  Double_Overlay [1] = htonl (Holding_Buffer);
95  } else {
96  return;
97  }
98 }
99 
100 // Float version
101 static void htonf (float &x)
102 {
103  if ( isLittleEndian ) {
104  int *Float_Overlay;
105  int Holding_Buffer;
106 
107  Float_Overlay = (int *) &x;
108  Holding_Buffer = Float_Overlay [0];
109 
110  Float_Overlay [0] = htonl (Holding_Buffer);
111  } else {
112  return;
113  }
114 }

149 void FGOutputFG::SocketDataFill(FGNetFDM* net)
150 {
151  unsigned int i;
152 
153  // Version
154  net->version = FG_NET_FDM_VERSION;
155 
156  // Positions
157  net->longitude = Propagate->GetLocation().GetLongitude(); //
158  net->latitude = Propagate->GetLocation().GetGeodLatitudeRad(); // geodetic (radians)
159  net->altitude = Propagate->GetAltitudeASL()*0.3048; // altitude, above sea level (meters)
160  net->agl = (float)(Propagate->GetDistanceAGL()*0.3048); // altitude, above ground level (meters)
161 
162  net->phi = (float)(Propagate->GetEuler(ePhi)); // roll (radians)
163  net->theta = (float)(Propagate->GetEuler(eTht)); // pitch (radians)
164  net->psi = (float)(Propagate->GetEuler(ePsi)); // yaw or true heading (radians)
165 
166  net->alpha = (float)(Auxiliary->Getalpha()); // angle of attack (radians)
167  net->beta = (float)(Auxiliary->Getbeta()); // side slip angle (radians)
168 
169  // Velocities
170  net->phidot = (float)(Auxiliary->GetEulerRates(ePhi)); // roll rate (radians/sec)
171  net->thetadot = (float)(Auxiliary->GetEulerRates(eTht)); // pitch rate (radians/sec)
172  net->psidot = (float)(Auxiliary->GetEulerRates(ePsi)); // yaw rate (radians/sec)
173  net->vcas = (float)(Auxiliary->GetVcalibratedKTS()); // VCAS, knots
174  net->climb_rate = (float)(Propagate->Gethdot()); // altitude rate, ft/sec
175  net->v_north = (float)(Propagate->GetVel(eNorth)); // north vel in NED frame, fps
176  net->v_east = (float)(Propagate->GetVel(eEast)); // east vel in NED frame, fps
177  net->v_down = (float)(Propagate->GetVel(eDown)); // down vel in NED frame, fps
178 //---ADD METHOD TO CALCULATE THESE TERMS---
179  net->v_body_u = (float)(Propagate->GetUVW(1)); // ECEF speed in body axis
180  net->v_body_v = (float)(Propagate->GetUVW(2)); // ECEF speed in body axis
181  net->v_body_w = (float)(Propagate->GetUVW(3)); // ECEF speed in body axis
182 
183  // Accelerations
184  net->A_X_pilot = (float)(Auxiliary->GetPilotAccel(1)); // X body accel, ft/s/s
185  net->A_Y_pilot = (float)(Auxiliary->GetPilotAccel(2)); // Y body accel, ft/s/s
186  net->A_Z_pilot = (float)(Auxiliary->GetPilotAccel(3)); // Z body accel, ft/s/s
187 
188  // Stall
189  net->stall_warning = 0.0; // 0.0 - 1.0 indicating the amount of stall
190  net->slip_deg = (float)(Auxiliary->Getbeta(inDegrees)); // slip ball deflection, deg
191 
192  net->num_engines = min(FGNetFDM::FG_MAX_ENGINES,Propulsion->GetNumEngines()); // Number of valid engines
193 
194  for (i=0; i<net->num_engines; i++) {
195  if (Propulsion->GetEngine(i)->GetRunning())
196  net->eng_state[i] = 2; // Engine state running
197  else if (Propulsion->GetEngine(i)->GetCranking())
198  net->eng_state[i] = 1; // Engine state cranking
199  else
200  net->eng_state[i] = 0; // Engine state off
201 
202  switch (Propulsion->GetEngine(i)->GetType()) {
203  case (FGEngine::etRocket):
204  break;
205  case (FGEngine::etPiston):
206  net->rpm[i] = (float)(((FGPiston *)Propulsion->GetEngine(i))->getRPM());
207  net->fuel_flow[i] = (float)(((FGPiston *)Propulsion->GetEngine(i))->getFuelFlow_gph());
208  net->fuel_px[i] = 0; // Fuel pressure, psi (N/A in current model)
209  net->egt[i] = (float)(((FGPiston *)Propulsion->GetEngine(i))->GetEGT());
210  net->cht[i] = (float)(((FGPiston *)Propulsion->GetEngine(i))->getCylinderHeadTemp_degF());
211  net->mp_osi[i] = (float)(((FGPiston *)Propulsion->GetEngine(i))->getManifoldPressure_inHg());
212  net->oil_temp[i] = (float)(((FGPiston *)Propulsion->GetEngine(i))->getOilTemp_degF());
213  net->oil_px[i] = (float)(((FGPiston *)Propulsion->GetEngine(i))->getOilPressure_psi());
214  net->tit[i] = 0; // Turbine Inlet Temperature (N/A for piston)
215  break;
216  case (FGEngine::etTurbine):
217  break;
218  case (FGEngine::etTurboprop):
219  break;
220  case (FGEngine::etElectric):
221  break;
222  case (FGEngine::etUnknown):
223  break;
224  }
225  }
226 
227  net->num_tanks = min(FGNetFDM::FG_MAX_TANKS, Propulsion->GetNumTanks()); // Max number of fuel tanks
228 
229  for (i=0; i<net->num_tanks; i++) {
230  net->fuel_quantity[i] = (float)(((FGTank *)Propulsion->GetTank(i))->GetContents());
231  }
232 
233  net->num_wheels = min(FGNetFDM::FG_MAX_WHEELS, GroundReactions->GetNumGearUnits());
234 
235  for (i=0; i<net->num_wheels; i++) {
236  net->wow[i] = GroundReactions->GetGearUnit(i)->GetWOW();
237  if (GroundReactions->GetGearUnit(i)->GetGearUnitDown())
238  net->gear_pos[i] = 1; //gear down, using FCS convention
239  else
240  net->gear_pos[i] = 0; //gear up, using FCS convention
241  net->gear_steer[i] = (float)(GroundReactions->GetGearUnit(i)->GetSteerNorm());
242  net->gear_compression[i] = (float)(GroundReactions->GetGearUnit(i)->GetCompLen());
243  }
244 
245  // Environment
246  net->cur_time = (long int)1234567890; // Friday, Feb 13, 2009, 23:31:30 UTC (not processed by FGFS anyway)
247  net->warp = 0; // offset in seconds to unix time
248  net->visibility = 25000.0; // visibility in meters (for env. effects)
249 
250  // Control surface positions (normalized values)
251  net->elevator = (float)(FCS->GetDePos(ofNorm)); // Norm Elevator Pos, --
252  net->elevator_trim_tab = (float)(FCS->GetPitchTrimCmd()); // Norm Elev Trim Tab Pos, --
253  net->left_flap = (float)(FCS->GetDfPos(ofNorm)); // Norm Flap Pos, --
254  net->right_flap = (float)(FCS->GetDfPos(ofNorm)); // Norm Flap Pos, --
255  net->left_aileron = (float)(FCS->GetDaLPos(ofNorm)); // Norm L Aileron Pos, --
256  net->right_aileron = (float)(FCS->GetDaRPos(ofNorm)); // Norm R Aileron Pos, --
257  net->rudder = (float)(FCS->GetDrPos(ofNorm)); // Norm Rudder Pos, --
258  net->nose_wheel = (float)(FCS->GetDrPos(ofNorm)); // *** FIX *** Using Rudder Pos for NWS, --
259  net->speedbrake = (float)(FCS->GetDsbPos(ofNorm)); // Norm Speedbrake Pos, --
260  net->spoilers = (float)(FCS->GetDspPos(ofNorm)); // Norm Spoiler Pos, --
261 
262  // Convert the net buffer to network format
263  if ( isLittleEndian ) {
264  net->version = htonl(net->version);
265 
266  htond(net->longitude);
267  htond(net->latitude);
268  htond(net->altitude);
269  htonf(net->agl);
270  htonf(net->phi);
271  htonf(net->theta);
272  htonf(net->psi);
273  htonf(net->alpha);
274  htonf(net->beta);
275 
276  htonf(net->phidot);
277  htonf(net->thetadot);
278  htonf(net->psidot);
279  htonf(net->vcas);
280  htonf(net->climb_rate);
281  htonf(net->v_north);
282  htonf(net->v_east);
283  htonf(net->v_down);
284  htonf(net->v_body_u);
285  htonf(net->v_body_v);
286  htonf(net->v_body_w);
287 
288  htonf(net->A_X_pilot);
289  htonf(net->A_Y_pilot);
290  htonf(net->A_Z_pilot);
291 
292  htonf(net->stall_warning);
293  htonf(net->slip_deg);
294 
295  for (i=0; i<net->num_engines; ++i ) {
296  net->eng_state[i] = htonl(net->eng_state[i]);
297  htonf(net->rpm[i]);
298  htonf(net->fuel_flow[i]);
299  htonf(net->fuel_px[i]);
300  htonf(net->egt[i]);
301  htonf(net->cht[i]);
302  htonf(net->mp_osi[i]);
303  htonf(net->tit[i]);
304  htonf(net->oil_temp[i]);
305  htonf(net->oil_px[i]);
306  }
307  net->num_engines = htonl(net->num_engines);
308 
309  for (i=0; i<net->num_tanks; ++i ) {
310  htonf(net->fuel_quantity[i]);
311  }
312  net->num_tanks = htonl(net->num_tanks);
313 
314  for (i=0; i<net->num_wheels; ++i ) {
315  net->wow[i] = htonl(net->wow[i]);
316  htonf(net->gear_pos[i]);
317  htonf(net->gear_steer[i]);
318  htonf(net->gear_compression[i]);
319  }
320  net->num_wheels = htonl(net->num_wheels);
321 
322  net->cur_time = htonl( net->cur_time );
323  net->warp = htonl( net->warp );
324  htonf(net->visibility);
325 
326  htonf(net->elevator);
327  htonf(net->elevator_trim_tab);
328  htonf(net->left_flap);
329  htonf(net->right_flap);
330  htonf(net->left_aileron);
331  htonf(net->right_aileron);
332  htonf(net->rudder);
333  htonf(net->nose_wheel);
334  htonf(net->speedbrake);
335  htonf(net->spoilers);
336  }
337 }
