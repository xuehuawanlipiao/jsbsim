219  // This list of enums is very important! The order in which models are listed
220  // here determines the order of execution of the models.
221  //
222  // There are some conditions that need to be met :
223  // 1. FCS can request mass geometry changes via the inertia/pointmass-*
224  // properties so it must be executed before MassBalance
225  // 2. MassBalance must be executed before Propulsion, Aerodynamics,
226  // GroundReactions, ExternalReactions and BuoyantForces to ensure that
227  // their moments are computed with the updated CG position.
228  enum eModels { ePropagate=0,
229  eInput,
230  eInertial,
231  eAtmosphere,
232  eWinds,
233  eSystems,
234  eMassBalance,
235  eAuxiliary,
