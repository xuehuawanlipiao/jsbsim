83 void FGOutputSocket::SetOutputName(const string& fname)
84 {
85  // tokenize the output name
86  size_t dot_pos = fname.find(':', 0);
87  size_t slash_pos = fname.find('/', 0);
88 
89  string name = fname.substr(0, dot_pos);
90 
91  string proto = "TCP";
92  if(dot_pos + 1 < slash_pos)
93  proto = fname.substr(dot_pos + 1, slash_pos - dot_pos - 1);
94 
95  string port = "1138";
96  if(slash_pos < string::npos)
97  port = fname.substr(slash_pos + 1, string::npos);
98 
99  // set the model name
100  Name = name + ":" + port + "/" + proto;
101 
102  // set the socket params
103  SockName = name;
104 
105  SockPort = atoi(port.c_str());
106 
107  if (to_upper(proto) == "UDP")
108  SockProtocol = FGfdmSocket::ptUDP;
109  else // Default to TCP
110  SockProtocol = FGfdmSocket::ptTCP;
111 }

129 bool FGOutputSocket::InitModel(void)
130 {
131  if (FGOutputType::InitModel()) {
132  delete socket;
133  socket = new FGfdmSocket(SockName, SockPort, SockProtocol);

138  PrintHeaders();

140  return true;

142

263  socket->Append(OutputCaptions[i]);

267

318  socket->Append(Aerodynamics->GetvFw()(eLift));

327  socket->Append(Aircraft->GetMoments(eN));
328  }
329  if (SubSystems & ssAtmosphere) {
330  socket->Append(Atmosphere->GetDensity());
331  socket->Append(Atmosphere->GetPressureSL());
332  socket->Append(Atmosphere->GetPressure());
333  socket->Append(Winds->GetTurbMagnitude());
334  socket->Append(Winds->GetTurbDirection());

336  }
337  if (SubSystems & ssMassProps) {

351  }
352  if (SubSystems & ssPropagate) {
353  socket->Append(Propagate->GetAltitudeASL());
354  socket->Append(radtodeg*Propagate->GetEuler(ePhi));
355  socket->Append(radtodeg*Propagate->GetEuler(eTht));
356  socket->Append(radtodeg*Propagate->GetEuler(ePsi));
357  socket->Append(Auxiliary->Getalpha(inDegrees));
358  socket->Append(Auxiliary->Getbeta(inDegrees));
359  socket->Append(Propagate->GetLocation().GetLatitudeDeg());
360  socket->Append(Propagate->GetLocation().GetLongitudeDeg());
361  }
362  if (SubSystems & ssAeroFunctions) {
363  scratch = Aerodynamics->GetAeroFunctionValues(",");
364  if (scratch.length() != 0) socket->Append(scratch);
365  }
366  if (SubSystems & ssFCS) {
367  scratch = FCS->GetComponentValues(",");
368  if (scratch.length() != 0) socket->Append(scratch);
369  }
370  if (SubSystems & ssGroundReactions) {
371  socket->Append(GroundReactions->GetGroundReactionValues(","));
372  }
373  if (SubSystems & ssPropulsion && Propulsion->GetNumEngines() > 0) {
374  socket->Append(Propulsion->GetPropulsionValues(","));
375  }
376 
377  for (unsigned int i=0;i<OutputParameters.size();++i) {
378  socket->Append(OutputParameters[i]->GetValue());
379  }
380 
381  socket->Send();
382 }
